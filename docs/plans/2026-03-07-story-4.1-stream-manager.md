```go
package manager

import (
	"context"
	"errors"
	"fmt"

	"go.mongodb.org/mongo-driver/v2/bson"
	"go.mongodb.org/mongo-driver/v2/mongo"
	"go.mongodb.org/mongo-driver/v2/mongo/options"
	"go.uber.org/zap"
	"google.golang.org/protobuf/types/known/timestamppb"

	streamlinev1 "github.com/alexrybrown/streamline/gen/go/streamline/v1"
)

// ErrStreamNotFound is returned when a stream does not exist in the store.
var ErrStreamNotFound = errors.New("stream not found")

// collectionName is the MongoDB collection for stream documents.
const collectionName = "streams"

// StoreConfig holds configuration for the MongoDB stream store.
type StoreConfig struct {
	Database *mongo.Database
	Log      *zap.Logger
}

// Store persists stream records in MongoDB.
type Store struct {
	collection *mongo.Collection
	log        *zap.Logger
}

// streamDocument is the MongoDB document representation of a Stream.
// Protobuf messages are not directly BSON-serializable, so we map to/from
// this struct at the persistence boundary.
type streamDocument struct {
	ID              string                       `bson:"_id"`
	SourceURI       string                       `bson:"source_uri"`
	State           int32                        `bson:"state"`
	EncodingProfile *encodingProfileDocument     `bson:"encoding_profile,omitempty"`
	CreatedAt       int64                        `bson:"created_at"`
	UpdatedAt       int64                        `bson:"updated_at"`
}

type encodingProfileDocument struct {
	Codec            string `bson:"codec"`
	Width            int32  `bson:"width"`
	Height           int32  `bson:"height"`
	BitrateKbps      int32  `bson:"bitrate_kbps"`
	Framerate        int32  `bson:"framerate"`
	SegmentDurationS int32  `bson:"segment_duration_s"`
}

// NewStore creates a new MongoDB stream store.
// A nil Log defaults to a no-op logger.
func NewStore(cfg StoreConfig) *Store {
	if cfg.Database == nil {
		panic("manager.StoreConfig: Database must not be nil")
	}
	if cfg.Log == nil {
		cfg.Log = zap.NewNop()
	}

	collection := cfg.Database.Collection(collectionName)

	// Create a unique index on source_uri for idempotent lookups.
	// Use a background context since this runs at startup.
	indexModel := mongo.IndexModel{
		Keys:    bson.D{{Key: "source_uri", Value: 1}},
		Options: options.Index().SetUnique(true),
	}
	if _, err := collection.Indexes().CreateOne(context.Background(), indexModel); err != nil {
		cfg.Log.Warn("failed to create source_uri index", zap.Error(err))
	}

	return &Store{
		collection: collection,
		log:        cfg.Log.Named("store"),
	}
}

// Create inserts a new stream document. It generates an ID, sets state to
// PROVISIONING, and populates timestamps. The stream is modified in place.
func (store *Store) Create(ctx context.Context, stream *streamlinev1.Stream) error {
	now := timestamppb.Now()
	stream.Id = bson.NewObjectID().Hex()
	stream.State = streamlinev1.StreamState_STREAM_STATE_PROVISIONING
	stream.CreatedAt = now
	stream.UpdatedAt = now

	document := toDocument(stream)
	if _, err := store.collection.InsertOne(ctx, document); err != nil {
		return fmt.Errorf("insert stream: %w", err)
	}
	return nil
}

// Get retrieves a stream by ID. Returns ErrStreamNotFound if not found.
func (store *Store) Get(ctx context.Context, streamID string) (*streamlinev1.Stream, error) {
	var document streamDocument
	err := store.collection.FindOne(ctx, bson.D{{Key: "_id", Value: streamID}}).Decode(&document)
	if errors.Is(err, mongo.ErrNoDocuments) {
		return nil, ErrStreamNotFound
	}
	if err != nil {
		return nil, fmt.Errorf("find stream: %w", err)
	}
	return fromDocument(&document), nil
}

// UpdateState transitions a stream to a new state. Returns ErrStreamNotFound
// if no document matches the given ID.
func (store *Store) UpdateState(ctx context.Context, streamID string, state streamlinev1.StreamState) error {
	now := timestamppb.Now()
	result, err := store.collection.UpdateOne(ctx,
		bson.D{{Key: "_id", Value: streamID}},
		bson.D{{Key: "$set", Value: bson.D{
			{Key: "state", Value: int32(state)},
			{Key: "updated_at", Value: now.AsTime().UnixNano()},
		}}},
	)
	if err != nil {
		return fmt.Errorf("update stream state: %w", err)
	}
	if result.MatchedCount == 0 {
		return ErrStreamNotFound
	}
	return nil
}

// FindBySourceURI looks up a stream by source URI. Returns nil, nil if not found.
func (store *Store) FindBySourceURI(ctx context.Context, sourceURI string) (*streamlinev1.Stream, error) {
	var document streamDocument
	err := store.collection.FindOne(ctx, bson.D{{Key: "source_uri", Value: sourceURI}}).Decode(&document)
	if errors.Is(err, mongo.ErrNoDocuments) {
		return nil, nil
	}
	if err != nil {
		return nil, fmt.Errorf("find stream by source URI: %w", err)
	}
	return fromDocument(&document), nil
}

func toDocument(stream *streamlinev1.Stream) *streamDocument {
	document := &streamDocument{
		ID:        stream.Id,
		SourceURI: stream.SourceUri,
		State:     int32(stream.State),
		CreatedAt: stream.CreatedAt.AsTime().UnixNano(),
		UpdatedAt: stream.UpdatedAt.AsTime().UnixNano(),
	}
	if profile := stream.EncodingProfile; profile != nil {
		document.EncodingProfile = &encodingProfileDocument{
			Codec:            profile.Codec,
			Width:            profile.Width,
			Height:           profile.Height,
			BitrateKbps:      profile.BitrateKbps,
			Framerate:        profile.Framerate,
			SegmentDurationS: profile.SegmentDurationS,
		}
	}
	return document
}

func fromDocument(document *streamDocument) *streamlinev1.Stream {
	stream := &streamlinev1.Stream{
		Id:        document.ID,
		SourceUri: document.SourceURI,
		State:     streamlinev1.StreamState(document.State),
		CreatedAt: &timestamppb.Timestamp{Seconds: document.CreatedAt / 1e9, Nanos: int32(document.CreatedAt % 1e9)},
		UpdatedAt: &timestamppb.Timestamp{Seconds: document.UpdatedAt / 1e9, Nanos: int32(document.UpdatedAt % 1e9)},
	}
	if profile := document.EncodingProfile; profile != nil {
		stream.EncodingProfile = &streamlinev1.EncodingProfile{
			Codec:            profile.Codec,
			Width:            profile.Width,
			Height:           profile.Height,
			BitrateKbps:      profile.BitrateKbps,
			Framerate:        profile.Framerate,
			SegmentDurationS: profile.SegmentDurationS,
		}
	}
	return stream
}
```

```go
package manager

import (
	"context"
	"errors"

	"connectrpc.com/connect"
	"go.uber.org/zap"

	streamlinev1 "github.com/alexrybrown/streamline/gen/go/streamline/v1"
	"github.com/alexrybrown/streamline/gen/go/streamline/v1/streamlinev1connect"
)

// streamStore is the narrow interface for stream persistence.
// Defined in the consuming package per Option D convention.
type streamStore interface {
	Create(ctx context.Context, stream *streamlinev1.Stream) error
	Get(ctx context.Context, streamID string) (*streamlinev1.Stream, error)
	UpdateState(ctx context.Context, streamID string, state streamlinev1.StreamState) error
	FindBySourceURI(ctx context.Context, sourceURI string) (*streamlinev1.Stream, error)
}

// ServiceConfig holds configuration for the stream manager Service.
type ServiceConfig struct {
	Store streamStore
	Log   *zap.Logger
}

// Service implements the StreamManagerServiceHandler ConnectRPC interface.
// It manages stream lifecycle (start, stop, restart) with MongoDB persistence.
type Service struct {
	store streamStore
	log   *zap.Logger
}

// Compile-time assertion that Service implements StreamManagerServiceHandler.
var _ streamlinev1connect.StreamManagerServiceHandler = (*Service)(nil)

// NewService creates a new stream manager Service.
// A nil Log defaults to a no-op logger.
func NewService(cfg ServiceConfig) *Service {
	if cfg.Store == nil {
		panic("manager.ServiceConfig: Store must not be nil")
	}
	if cfg.Log == nil {
		cfg.Log = zap.NewNop()
	}
	return &Service{
		store: cfg.Store,
		log:   cfg.Log.Named("service"),
	}
}

// StartStream creates a new stream or returns an existing one for the same source URI.
func (service *Service) StartStream(ctx context.Context, request *connect.Request[streamlinev1.StartStreamRequest]) (*connect.Response[streamlinev1.StartStreamResponse], error) {
	log := service.log.With(
		zap.String("method", "StartStream"),
		zap.String("sourceURI", request.Msg.SourceUri),
	)

	// Idempotency: return existing stream if one exists for this source URI.
	existing, err := service.store.FindBySourceURI(ctx, request.Msg.SourceUri)
	if err != nil {
		log.Error("failed to find stream by source URI", zap.Error(err))
		return nil, connect.NewError(connect.CodeInternal, err)
	}
	if existing != nil {
		log.Info("returning existing stream", zap.String("streamID", existing.Id))
		return connect.NewResponse(&streamlinev1.StartStreamResponse{Stream: existing}), nil
	}

	stream := &streamlinev1.Stream{
		SourceUri:       request.Msg.SourceUri,
		EncodingProfile: request.Msg.EncodingProfile,
	}

	if err := service.store.Create(ctx, stream); err != nil {
		log.Error("failed to create stream", zap.Error(err))
		return nil, connect.NewError(connect.CodeInternal, err)
	}

	log.Info("stream created", zap.String("streamID", stream.Id))
	return connect.NewResponse(&streamlinev1.StartStreamResponse{Stream: stream}), nil
}

// StopStream transitions a stream to STOPPED state.
func (service *Service) StopStream(ctx context.Context, request *connect.Request[streamlinev1.StopStreamRequest]) (*connect.Response[streamlinev1.StopStreamResponse], error) {
	log := service.log.With(
		zap.String("method", "StopStream"),
		zap.String("streamID", request.Msg.StreamId),
	)

	if err := service.store.UpdateState(ctx, request.Msg.StreamId, streamlinev1.StreamState_STREAM_STATE_STOPPED); err != nil {
		if errors.Is(err, ErrStreamNotFound) {
			return nil, connect.NewError(connect.CodeNotFound, err)
		}
		log.Error("failed to update stream state", zap.Error(err))
		return nil, connect.NewError(connect.CodeInternal, err)
	}

	stream, err := service.store.Get(ctx, request.Msg.StreamId)
	if err != nil {
		log.Error("failed to get stream after stop", zap.Error(err))
		return nil, connect.NewError(connect.CodeInternal, err)
	}

	log.Info("stream stopped")
	return connect.NewResponse(&streamlinev1.StopStreamResponse{Stream: stream}), nil
}

// RestartStream stops then re-provisions a stream.
func (service *Service) RestartStream(ctx context.Context, request *connect.Request[streamlinev1.RestartStreamRequest]) (*connect.Response[streamlinev1.RestartStreamResponse], error) {
	log := service.log.With(
		zap.String("method", "RestartStream"),
		zap.String("streamID", request.Msg.StreamId),
	)

	// Transition to PROVISIONING (restart semantics: stop + re-provision).
	if err := service.store.UpdateState(ctx, request.Msg.StreamId, streamlinev1.StreamState_STREAM_STATE_PROVISIONING); err != nil {
		if errors.Is(err, ErrStreamNotFound) {
			return nil, connect.NewError(connect.CodeNotFound, err)
		}
		log.Error("failed to restart stream", zap.Error(err))
		return nil, connect.NewError(connect.CodeInternal, err)
	}

	stream, err := service.store.Get(ctx, request.Msg.StreamId)
	if err != nil {
		log.Error("failed to get stream after restart", zap.Error(err))
		return nil, connect.NewError(connect.CodeInternal, err)
	}

	log.Info("stream restarted")
	return connect.NewResponse(&streamlinev1.RestartStreamResponse{Stream: stream}), nil
}
```

Follow the pattern from `cmd/packager/main.go`: config, MongoDB client, service, HTTP server with h2c, graceful shutdown.

```go
package main

import (
	"context"
	"errors"
	"fmt"
	"net/http"
	"os"
	"os/signal"
	"syscall"
	"time"

	"connectrpc.com/connect"
	connectcors "connectrpc.com/cors"
	"connectrpc.com/grpcreflect"
	"github.com/rs/cors"
	"go.mongodb.org/mongo-driver/v2/mongo"
	"go.mongodb.org/mongo-driver/v2/mongo/options"
	"go.uber.org/zap"

	"github.com/alexrybrown/streamline/gen/go/streamline/v1/streamlinev1connect"
	"github.com/alexrybrown/streamline/internal/config"
	"github.com/alexrybrown/streamline/internal/health"
	"github.com/alexrybrown/streamline/internal/logging"
	"github.com/alexrybrown/streamline/internal/manager"
)

const (
	// defaultPort for the stream manager. 8080 matches config.Load() default.
	defaultPort = "8080"

	// mongoConnectTimeout is the maximum time to wait for MongoDB connection.
	mongoConnectTimeout = 10 * time.Second

	// httpShutdownTimeout is the maximum time for graceful HTTP shutdown.
	httpShutdownTimeout = 10 * time.Second

	// corsMaxAgeSeconds for preflight cache. 2 hours balances reducing
	// preflight requests with allowing config changes.
	corsMaxAgeSeconds = 7200
)

func main() {
	log := logging.New("stream-manager")
	cfg := config.Load()

	port := cfg.ServicePort
	if port == "" {
		port = defaultPort
	}

	// Connect to MongoDB.
	connectCtx, connectCancel := context.WithTimeout(context.Background(), mongoConnectTimeout)
	defer connectCancel()

	mongoClient, err := mongo.Connect(options.Client().ApplyURI(cfg.MongoURI))
	if err != nil {
		log.Fatal("failed to create mongo client", zap.Error(err))
	}
	defer func() {
		if disconnectErr := mongoClient.Disconnect(context.Background()); disconnectErr != nil {
			log.Error("failed to disconnect mongo", zap.Error(disconnectErr))
		}
	}()

	if err := mongoClient.Ping(connectCtx, nil); err != nil {
		log.Fatal("failed to ping mongo", zap.Error(err))
	}

	database := mongoClient.Database(cfg.MongoDBName)

	store := manager.NewStore(manager.StoreConfig{
		Database: database,
		Log:      log,
	})

	service := manager.NewService(manager.ServiceConfig{
		Store: store,
		Log:   log,
	})

	// Build HTTP handler.
	mux := http.NewServeMux()

	connectPath, connectHandler := streamlinev1connect.NewStreamManagerServiceHandler(service)
	mux.Handle(connectPath, connectHandler)

	reflector := grpcreflect.NewStaticReflector(streamlinev1connect.StreamManagerServiceName)
	reflectPathV1, reflectHandlerV1 := grpcreflect.NewHandlerV1(reflector)
	mux.Handle(reflectPathV1, reflectHandlerV1)
	reflectPathV1Alpha, reflectHandlerV1Alpha := grpcreflect.NewHandlerV1Alpha(reflector)
	mux.Handle(reflectPathV1Alpha, reflectHandlerV1Alpha)

	checker := health.NewChecker()
	checker.AddCheck("mongodb", func() error {
		pingCtx, pingCancel := context.WithTimeout(context.Background(), 2*time.Second)
		defer pingCancel()
		return mongoClient.Ping(pingCtx, nil)
	})
	mux.Handle("/healthz", checker.Handler())

	// CORS middleware.
	middleware := cors.New(cors.Options{
		AllowedOrigins: []string{"*"},
		AllowedMethods: connectcors.AllowedMethods(),
		AllowedHeaders: connectcors.AllowedHeaders(),
		ExposedHeaders: connectcors.ExposedHeaders(),
		MaxAge:         corsMaxAgeSeconds,
	})

	protocols := new(http.Protocols)
	protocols.SetHTTP1(true)
	protocols.SetUnencryptedHTTP2(true)

	address := ":" + port
	server := &http.Server{
		Addr:              address,
		Handler:           middleware.Handler(mux),
		Protocols:         protocols,
		ReadHeaderTimeout: httpShutdownTimeout,
	}

	// Graceful shutdown.
	ctx, cancel := context.WithCancel(context.Background())
	defer cancel()

	signals := make(chan os.Signal, 1)
	signal.Notify(signals, syscall.SIGINT, syscall.SIGTERM)
	go func() {
		<-signals
		log.Info("shutting down")
		cancel()
	}()

	log.Info("starting stream manager",
		zap.String("address", address),
		zap.String("mongoURI", cfg.MongoURI),
		zap.String("database", cfg.MongoDBName),
	)

	errChan := make(chan error, 1)
	go func() {
		if listenErr := server.ListenAndServe(); listenErr != nil && !errors.Is(listenErr, http.ErrServerClosed) {
			errChan <- fmt.Errorf("HTTP server error: %w", listenErr)
		}
		close(errChan)
	}()

	<-ctx.Done()

	shutdownCtx, shutdownCancel := context.WithTimeout(context.Background(), httpShutdownTimeout)
	defer shutdownCancel()

	if shutdownErr := server.Shutdown(shutdownCtx); shutdownErr != nil {
		log.Error("HTTP server shutdown error", zap.Error(shutdownErr))
	}

	if serverErr := <-errChan; serverErr != nil {
		log.Fatal("server error", zap.Error(serverErr))
	}
}
```
